\begin{frame}{This paper}
  \begin{itemize}
    \item \textbf{Fact 1.} Hedge funds are the marginal intermediaries between sovereign cash and futures markets, and the trade they run is not the modeled basis trade.
    \item \textbf{Fact 2.} The trade concentrates in cheap off-the-run bonds and is duration neutral but convexity exposed.
    \item \textbf{Fact 3.} Three dealers finance more than half of it.
    \item \textbf{Experiment.} Dealer funding shocks and the April 2025 volatility shock reveal the fragility.
  \end{itemize}
\end{frame}

\begin{frame}{Data and setting}
  \textbf{One dataset per leg of the trade}
  \begin{itemize}
    \item \textbf{Cash leg}\quad \todohl{do we need this? MTS?}
    \item \textbf{Repo leg}\quad SFTDS transaction data
    \item \textbf{Futures leg}\quad EMIR transaction data
  \end{itemize}
  \bigskip
  \textbf{Coverage}
  \begin{itemize}
    \item SFTDS and EMIR report every transaction that involves at least one euro area established entity.
    \item Euro area dealers are global banks, so the data cover both parts of EUR and USD markets.
    \item US coverage is naturally thinner. A trade between two non euro area entities, for example a US dealer facing an offshore fund, never enters the data.
  \end{itemize}
\end{frame}

\section{Measuring basis positions}
\begin{frame}{Measuring basis positions}
  \textbf{Three building blocks}
  \begin{itemize}
    \item \textbf{Bond day panel}\quad fitted yields, pricing errors, duration and convexity, with CTD, deliverable and on the run flags
    \item \textbf{Repo panel}\quad SFTDS positions by fund, bond and day
    \item \textbf{Futures panel}\quad EMIR positions by fund, contract and day, linked to bonds through the deliverable set
  \end{itemize}
  \smallskip
  The panels merge in any combination of fund, bond and date.
  \medskip

  \textbf{Classifying positions}
  \begin{itemize}
    \item A fund that is long the future and short the bond, or the reverse, runs a basis trade.
    \item Bond positions without an offsetting futures leg are directional.
  \end{itemize}
  \medskip
  \textbf{Validation}
  \begin{itemize}
    \item Compare positions and captured market share with the OFR hedge fund monitor.
    \item \todohl{still need ECB numbers as the benchmark for the euro leg}
  \end{itemize}
\end{frame}

\section{Anatomy of the trade}
\begin{frame}{Anatomy of the trade}
  \begin{itemize}
    \item Positions are concentrated in a handful of funds.
    \item Three dealers finance more than half of the trade.
    \item The bonds are neither cheapest to deliver nor on the run and show positive pricing errors in both jurisdictions.
    \item Funds are paid to hold what curve sensitive investors and dealers do not want, providing liquidity and convergence in the illiquid segment.
  \end{itemize}
\end{frame}

\section{Risk and return of the trade}
\begin{frame}{Risk and return of the trade}
  \begin{itemize}
    \item Positions are duration matched but leave a convexity gap that widens with rate volatility and the term premium.
    \item Returns decompose into convergence, carry and specialness net of financing and margin costs.
    \item This yields an ex ante exposure measure at the fund and bond level, the treatment intensity for the fragility tests.
  \end{itemize}

\section{The funding architecture}
\begin{frame}{The funding architecture}
  \begin{itemize}
    \item Repo terms for every fund and dealer pair, including rates, haircuts, tenor and rollover frequency.
    \item Relationship stability over time.
    \item Funding is concentrated, short term and relationship based.
  \end{itemize}

\section{Fragility through the dealer funding channel}
\begin{frame}{Fragility through the dealer funding channel}
  \begin{itemize}
    \item Khwaja and Mian design comparing the same fund across its dealers with fund$\times$time fixed effects.
    \item Recurring year end balance sheet contractions give power and April 2025 gives salience.
    \item Trace the shock from financing quantities to position cuts to which bonds are cut first, expecting the cheapest and least liquid to go first.
  \end{itemize}
\end{frame}

% ===========================================================================
\section{Fragility through the volatility shock}
\begin{frame}{Fragility through the volatility shock}
  \begin{itemize}
    \item April 2025 realized the convexity gap. Exposure measured ex ante gives the treatment intensity.
    \item \textbf{Cross market test.} Did funds with larger UST exposure cut their EGB positions more within the same bond and week?
    \item \textbf{Spillover.} Did prices and liquidity of the exposed bonds move?
  \end{itemize}
\end{frame}

\begin{frame}{The causal chain}
  \centering
  Dealer or US book shock\\[.5em]
  $\downarrow$\\[.5em]
  The fund's financing or capital contracts \hfill {\small (dealer funding channel)}\\[.5em]
  $\downarrow$\\[.5em]
  The fund adjusts positions in the other market \hfill {\small (cross market transmission)}\\[.5em]
  $\downarrow$\\[.5em]
  Bond prices and liquidity move with exposure \hfill {\small (spillover result)}
\end{frame}

\section{Implications}
\begin{frame}{Implications}
  \begin{itemize}
    \item The risk is not directional exposure but convexity and funding concentration in illiquid bonds.
    \item Current NBFI monitoring does not measure this, with consequences for the transparency and leverage debates.
  \end{itemize}
\end{frame}
